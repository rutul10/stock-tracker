# News Tab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a News tab that shows aggregated stock news from the user's watchlist + popular stocks, with an optional ticker filter to narrow to a single symbol.

**Architecture:** Two tasks: (1) update the store's `Tab` type and wire the new tab into `App.tsx`, (2) implement the `News.tsx` component. The component fetches popular symbols from `/api/screener/popular`, unions them with the watchlist from the store, then fires concurrent `GET /api/stock/{symbol}/news` requests (capped at 15 symbols). Articles are merged, deduplicated by URL, and sorted newest-first. Each article shows a ticker badge in the all-news view. A filter input at the top switches to single-symbol mode.

**Tech Stack:** React 18, TypeScript, axios, Zustand store, existing `NewsData` type from `hooks/useStockDetail.ts`

## Task 2: Implement the News Component

**Files:**
- Create: `frontend/src/components/News/News.tsx`

The component has two modes:
- **All-news mode** (filterSymbol is empty): fetches news for `union(watchlist, popularSymbols).slice(0, 15)`, merges articles with a `symbol` field, deduplicates by URL, sorts newest-first.
- **Filtered mode** (filterSymbol is non-empty): fetches news for that single symbol only, no ticker badges shown.

- [ ] **Step 1: Create the News.tsx file with types and helpers**

Create `frontend/src/components/News/News.tsx`:

```typescript
import { useEffect, useRef, useState } from 'react'
import axios from 'axios'
import { useAppStore } from '../../store'
import { Spinner } from '../shared/Spinner'
import { ErrorBanner } from '../shared/ErrorBanner'

interface Article {
  symbol: string
  headline: string
  summary: string
  source: string
  url: string
  datetime: number
}

function relativeTime(unixTs: number): string {
  if (!unixTs) return ''
  const diff = Math.floor(Date.now() / 1000 - unixTs)
  if (diff < 60) return 'just now'
  if (diff < 3600) return `${Math.floor(diff / 60)}m ago`
  if (diff < 86400) return `${Math.floor(diff / 3600)}h ago`
  return `${Math.floor(diff / 86400)}d ago`
}

async function fetchSymbolNews(symbol: string): Promise<Article[]> {
  const res = await axios.get(`/api/stock/${symbol}/news`)
  const articles = res.data?.articles ?? []
  return articles.map((a: Omit<Article, 'symbol'>) => ({ ...a, symbol }))
}

function dedupeByUrl(articles: Article[]): Article[] {
  const seen = new Set<string>()
  return articles.filter((a) => {
    if (!a.url || seen.has(a.url)) return false
    seen.add(a.url)
    return true
  })
}
```

- [ ] **Step 2: Add the News component function**

Append to `frontend/src/components/News/News.tsx`:

```typescript
export function News() {
  const { watchlist } = useAppStore()
  const [filterInput, setFilterInput] = useState('')
  const [filterSymbol, setFilterSymbol] = useState('')
  const [articles, setArticles] = useState<Article[]>([])
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const filterRef = useRef<HTMLInputElement>(null)

  async function loadAll() {
    setLoading(true)
    setError(null)
    try {
      const popularRes = await axios.get('/api/screener/popular')
      const popularSymbols: string[] = (popularRes.data?.results ?? []).map(
        (r: { symbol: string }) => r.symbol
      )
      const combined = [...new Set([...watchlist, ...popularSymbols])].slice(0, 15)
      if (combined.length === 0) {
        setArticles([])
        return
      }
      const results = await Promise.allSettled(combined.map(fetchSymbolNews))
      const all: Article[] = results.flatMap((r) =>
        r.status === 'fulfilled' ? r.value : []
      )
      setArticles(
        dedupeByUrl(all).sort((a, b) => b.datetime - a.datetime)
      )
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : 'Failed to load news')
    } finally {
      setLoading(false)
    }
  }

  async function loadForSymbol(symbol: string) {
    setLoading(true)
    setError(null)
    try {
      const fetched = await fetchSymbolNews(symbol)
      setArticles(fetched.sort((a, b) => b.datetime - a.datetime))
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : `Failed to load news for ${symbol}`)
    } finally {
      setLoading(false)
    }
  }

  useEffect(() => {
    if (filterSymbol) {
      loadForSymbol(filterSymbol)
    } else {
      loadAll()
    }
  }, [filterSymbol, watchlist.join(',')])

  function handleFilterSubmit() {
    const sym = filterInput.trim().toUpperCase()
    setFilterSymbol(sym)
  }

  function handleClear() {
    setFilterInput('')
    setFilterSymbol('')
  }

  const showBadge = !filterSymbol

  return (
    <div style={{ padding: '16px', maxWidth: 860 }}>
      {/* Header */}
      <div style={{ display: 'flex', alignItems: 'center', gap: 16, marginBottom: 16, flexWrap: 'wrap' }}>
        <h2 style={{ fontFamily: 'var(--font-heading)', fontSize: '1.8rem', color: 'var(--accent-green)', letterSpacing: '0.1em', margin: 0 }}>
          NEWS
        </h2>
        <span style={{ color: 'var(--text-muted)', fontSize: 11, fontFamily: 'var(--font-ui)' }}>
          {filterSymbol
            ? `Showing news for ${filterSymbol}`
            : `Watchlist + popular · ${articles.length} articles`}
        </span>

        {/* Filter */}
        <div style={{ marginLeft: 'auto', display: 'flex', gap: 6, alignItems: 'center' }}>
          <input
            ref={filterRef}
            value={filterInput}
            onChange={(e) => setFilterInput(e.target.value.toUpperCase())}
            onKeyDown={(e) => {
              if (e.key === 'Enter') handleFilterSubmit()
              if (e.key === 'Escape') handleClear()
            }}
            placeholder="FILTER BY TICKER"
            style={{
              width: 160,
              background: 'var(--surface)',
              color: 'var(--text-primary)',
              border: `1px solid ${filterSymbol ? 'var(--accent-blue)' : 'var(--border)'}`,
              padding: '5px 10px',
              fontFamily: 'var(--font-mono)',
              fontSize: 12,
              letterSpacing: '0.06em',
            }}
          />
          <button
            onClick={handleFilterSubmit}
            disabled={!filterInput.trim()}
            style={{
              background: filterInput.trim() ? 'var(--accent-blue)' : 'var(--border)',
              color: filterInput.trim() ? '#000' : 'var(--text-muted)',
              border: 'none',
              padding: '5px 12px',
              fontFamily: 'var(--font-ui)',
              fontSize: 11,
              fontWeight: 700,
              cursor: filterInput.trim() ? 'pointer' : 'default',
              letterSpacing: '0.06em',
            }}
          >
            FILTER
          </button>
          {filterSymbol && (
            <button
              onClick={handleClear}
              style={{
                background: 'none',
                border: '1px solid var(--border)',
                color: 'var(--text-muted)',
                padding: '5px 10px',
                fontFamily: 'var(--font-ui)',
                fontSize: 11,
                cursor: 'pointer',
                letterSpacing: '0.06em',
              }}
            >
              CLEAR
            </button>
          )}
          <button
            onClick={() => filterSymbol ? loadForSymbol(filterSymbol) : loadAll()}
            disabled={loading}
            style={{
              background: 'none',
              border: '1px solid var(--border)',
              color: 'var(--text-muted)',
              padding: '5px 10px',
              fontFamily: 'var(--font-ui)',
              fontSize: 11,
              cursor: loading ? 'not-allowed' : 'pointer',
            }}
          >
            {loading ? '...' : '↻'}
          </button>
        </div>
      </div>

      {error && <ErrorBanner message={error} />}

      {loading && articles.length === 0 && <Spinner />}

      {!loading && articles.length === 0 && !error && (
        <div style={{ color: 'var(--text-muted)', fontFamily: 'var(--font-ui)', fontSize: 12, padding: '40px', textAlign: 'center', border: '1px solid var(--border)' }}>
          {filterSymbol
            ? `No recent news found for ${filterSymbol}`
            : 'No news found. Add symbols to your watchlist or wait for popular stocks to load.'}
        </div>
      )}

      {/* Article list */}
      <div style={{ display: 'flex', flexDirection: 'column', gap: 0 }}>
        {articles.map((article, i) => (
          <div
            key={`${article.url}-${i}`}
            style={{
              borderBottom: '1px solid var(--border)',
              padding: '12px 4px',
              display: 'flex',
              gap: 12,
              alignItems: 'flex-start',
            }}
          >
            {/* Ticker badge */}
            {showBadge && (
              <span style={{
                flexShrink: 0,
                fontFamily: 'var(--font-mono)',
                fontSize: 10,
                fontWeight: 700,
                color: 'var(--accent-blue)',
                background: 'rgba(0,170,255,0.1)',
                border: '1px solid rgba(0,170,255,0.25)',
                padding: '2px 6px',
                letterSpacing: '0.05em',
                marginTop: 2,
                minWidth: 52,
                textAlign: 'center',
              }}>
                {article.symbol}
              </span>
            )}

            <div style={{ flex: 1, minWidth: 0 }}>
              {/* Headline */}
              <a
                href={article.url}
                target="_blank"
                rel="noopener noreferrer"
                style={{
                  color: 'var(--text-primary)',
                  textDecoration: 'none',
                  fontFamily: 'var(--font-ui)',
                  fontSize: 13,
                  lineHeight: 1.45,
                  display: 'block',
                  marginBottom: 4,
                }}
                onMouseEnter={(e) => (e.currentTarget.style.color = 'var(--accent-blue)')}
                onMouseLeave={(e) => (e.currentTarget.style.color = 'var(--text-primary)')}
              >
                {article.headline}
              </a>

              {/* Summary */}
              {article.summary && article.summary !== article.headline && (
                <p style={{
                  margin: '0 0 6px',
                  color: 'var(--text-muted)',
                  fontFamily: 'var(--font-ui)',
                  fontSize: 12,
                  lineHeight: 1.5,
                  display: '-webkit-box',
                  WebkitLineClamp: 2,
                  WebkitBoxOrient: 'vertical',
                  overflow: 'hidden',
                }}>
                  {article.summary}
                </p>
              )}

              {/* Meta */}
              <div style={{ display: 'flex', gap: 10, fontSize: 10, color: 'var(--text-muted)', fontFamily: 'var(--font-ui)' }}>
                {article.source && <span>{article.source}</span>}
                {article.datetime > 0 && <span>{relativeTime(article.datetime)}</span>}
              </div>
            </div>
          </div>
        ))}
      </div>
    </div>
  )
}
```
